  CREATE TABLE Article (
    ArticleId INT PRIMARY KEY,
    Date DATETIME,
    Author VARCHAR(255),
    Title VARCHAR(255),
    Year VARCHAR(255),
    Journal VARCHAR(255),
    Volume VARCHAR(255),
    Issue VARCHAR(255),
    DOI VARCHAR(255),
    URL VARCHAR(255),
    DB VARCHAR(255),
    language VARCHAR(255),
    keyword VARCHAR(255)
  );
